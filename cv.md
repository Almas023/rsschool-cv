* **Get the Middle Character**
```function getMiddle(s) {
  const l = Math.floor(s.length/2);
  let result = "";
    if (s.length % 2 == 0) {
      for (let i=(l-1); i<=l; i++)
        result += s[i];
    } else if (s.length % 2 != 0) {
      result += s[l];
    } else {
      return s;
    }
  return result;
}
```

* **Highest and Lowest**
```function highAndLow(numbers){
  const s = numbers.split(' ');
  const srt = s.sort((a,b) => {
    return b - a;
  })
  
  const max = srt[0];
  const min = srt[srt.length - 1];
   
  return max + " " + min;
}
```
